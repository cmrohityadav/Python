# Object Types / Data Types
- Numbers : 123, 3.15, 2+4j, 0b11, Decimal(), Fraction(), 
- String : "spam", 'rohit',b'a\x01c', u'sp\xc4m',

- Tuple : (1, 'spam', 4, 'U'), tuple('spam'), namedtuple

- Set : set('abc'), {'a', 'b', 'c'}

- File : open('eggs.txt'), open(r'C:\ham.bin', 'wb')

- Boolean : True, False
- None : None
